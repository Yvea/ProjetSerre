humidCritique et humidNecessaire

alt humidSol < humidCritique
    GestionSerre->Actionneur: this.arrosage.Commander('on')
else humidSol > humidNecessaire
    GestionSerre->Actionneur: this.arrosage.Commander('of')
else 
    GestionSerre->Actionneur: this.arrosage.Commander('on')
